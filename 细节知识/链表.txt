struct node* creat()
{
struct node *head,*New,*tail;
head=NULL;
int n=0;
printf("输入整数，-1为结束标志\n");
 scanf("%d",&n);
while(n!=-1)
{
    New=(struct node*)malloc(sizeof(struct node));
    New->number=n;
    New->next=NULL;
    if(head==NULL)
        head=New;
    else
        tail->next=New;
    tail=New;
    scanf("%d",&n);
}
fflush(stdin);
return head;
}
